  capEvent:(frame)=>
    @url = frame.data.imgfile
    @imgcurr=(@imgcurr+1)%@imglen
    ci = $(@imgs[@imgcurr])
    ci.attr('src',@url)
    return
    
  render:=>

      o.onload = @loaderToggle

    @imglen = @imgs.length
    @imgcurr = 0
